heat_template_version: 2013-05-23

description: 1 web server deployment template

parameters:
  key_name:
    type: string
    description: Name of our keypair
    default: simple-python-server
  
  flavor:
    type: string
    description: Flavor for the server to be created
    default: INF4410-mini
  
  image:
    type: string
    description: Image name
    default: INF4410-Ubuntu-trusty-mini
  
  network_id:
    type: string
    description: network to be used
    default: 0c3b1354-d3d5-4fec-86d3-19d4d1f13785

  server_url:
    type: string
    description: URL containing the python server code
    default: https://raw.githubusercontent.com/houssemmh/INF8480-TP3/master/server.py

resources:
  server:
    type: OS::Nova::Server
    properties:
      key_name: { get_param: key_name }
      image: { get_param: image }
      flavor: { get_param: flavor }
      networks:
        - network: { get_param: network_id }
      user_data_format: RAW
      user_data:
        str_replace:
          template: |
            #!/bin/bash
            curl server_url -o server.py -s;
            python server.py;
          params:
            server_url: { get_param: server_url }

outputs:
  server_networks:
    description: The networks of the deployed server
    value: { get_attr: [server_url, network_id] }
